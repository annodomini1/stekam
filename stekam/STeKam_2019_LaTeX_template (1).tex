\begin{equation}
||\textbf{p}||^2=(a\textbf{p})^Ta\textbf{p}, \ \ pogoj: \ \ \textbf{p}^T\textbf{p}=1
\end{equation}

\begin{equation}
L(\textbf{p}) = \textbf{p}^T a^T a \textbf{p} - \lambda(\textbf{p}^T \textbf{p} - 1)
\end{equation}
\begin{equation}
\label{eq_min}
a^T a \textbf{p} - \lambda \textbf{p} = 0
\end{equation}

V izrazu \ref{eq_min} je torej $\lambda$ lastna vrednost matrike $a^Ta$, ki ji pripada lastni vektor $\textbf{p} = \textbf{e}_\lambda$. Ta nam da $L(\textbf{e}_\lambda) = \lambda$, ki ima najmanjšo vrednost pri $\lambda = 0$.


\subsection{Filtrirana povratna projekcija}

Pomagamo si z \cite{spic1}. Jedro naloge predstavlja postopek FBP, ki je eden od temelnjih naèinov rekonstrukcije pri slikanju z raèunalniško tomografijo na podroèju zdravstva. \begin{equation}
x \cos\theta + y \sin \theta = r
\end{equation}

\begin{equation}
p_{\theta}(r) = \int_{-\infty}^{\infty} \int_{-\infty}^{\infty} f(x, y) \delta (x \cos\theta + y \sin \theta - r)dx dy
\end{equation}

\begin{figure}[H]
	\centerline{\includegraphics[width=5.5cm]{fig/FBPderi}}
	\caption{Filtrirana povratna projekcija.}
	\label{fig:fbp_deri}
\end{figure}

Zajete projekcije so nato pretvorjene v frekvenèno domeno s pomoèjo Fourier-ove transformacije $P_{\theta}(\omega)$ za vsak kot. Temu nato sledi filtriranje projekcij s frekvenènim odzivom $|\omega|$, kar opisuje izraz \ref{ge}.

\begin{equation}
P_{\theta}(\omega) = \frac{1}{\sqrt{2\pi}} \int_{-\infty}^{\infty} p_{\theta}(r) e^{-i \omega r} dr
\end{equation}

\begin{equation}
\label{ge}
g_{\theta}(r) = \int_{-\infty}^{\infty}  P_{\theta}(\omega)|\omega|e^{i 2\pi\omega r} d\omega
\end{equation}

S filtriranjem se želimo znebiti visokih frekvenc v projekcijah in posledièno pridobiti kvalitetnejšo rekonstrukcijo. Po filtriranju nato sledi še rekonstrukcija v $f'(x,y)$ in sicer tako, da filtrirane inverze Fourier-ove transformacije seštejemo po pripadajoèih kotih in posledièno dobimo rekonstruiran objekt.

\begin{equation}
f'(x,y) = \frac{1}{2\pi} \int_{0}^{\pi} g_{\theta}(r) (x \cos\theta + y \sin \theta)d\theta
\end{equation}

Ali v diskretni obliki, kjer je $\Delta\theta$ korak diskretizacije:

\begin{equation}
f'(x,y) = \frac{1}{2\pi} \sum\limits_{i=0}^{N-1} \Delta\theta_i g_{\theta_i}(x \cos\theta_i + \sin\theta_i)
\end{equation}
